554     /**
555      * Converts a DOS timestamp to UNIX format.There are very few cases where
556      * this is needed, but some binary formats use it (eg: zip files.)
557      * Converting the other direction is done using {@link Date::unix2dos}.
558      *
559      *     $unix = Date::dos2unix($dos);
560      *
561      * @param   integer  DOS timestamp
562      * @return  integer
563      */
564     public static function dos2unix($timestamp = FALSE)
565     {
566         $sec  = 2 * ($timestamp & 0x1f);
567         $min  = ($timestamp >>  5) & 0x3f;
568         $hrs  = ($timestamp >> 11) & 0x1f;
569         $day  = ($timestamp >> 16) & 0x1f;
570         $mon  = ($timestamp >> 21) & 0x0f;
571         $year = ($timestamp >> 25) & 0x7f;
572 
573         return mktime($hrs, $min, $sec, $mon, $day, $year + 1980);
574     }
