564      *
565      * @param   mixed   table name or array(table, alias)

569      */

571     {

574             list($table, $alias) = $table;
575         }
576 
577         if ($table instanceof Database_Query)
578         {
579             // Create a sub-query

581         }
582         elseif ($table instanceof Database_Expression)

585             $table = $table->value();
586         }
587         else

591

595

600

602                     $parts[$offset] = $prefix.$parts[$offset];

604

606                 {

610

612             }

617             }

621         {
